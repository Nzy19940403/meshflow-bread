# 面包店经营模型 V5.3 — 完整数学建模
## 基于中国三线城市真实面包店数据校准 (2024)

=============================================================================
## 一、状态变量 (State)
=============================================================================

c = {
  B1:  售价                  (滑条 ¥5~40)
  B2:  市场需求/订单数       (计算量)
  B3:  实际产量               (计算量)
  B4:  报废/加工成本/份      (¥, 初始2.0, 渐降至~1.0)
  B5:  月租金                 (¥)
  B6:  月销量 = min(B2, B3)   (份)
  B7:  月营收                 (¥, 含饮品)
  B8:  月利润                 (¥)
  B9:  月人工成本             (¥)
  B10: 原料进价               (滑条 ¥1~6, 默认 5)
  B11: 单位变动成本           (¥)
  B12: 实际原料成本/份       (¥, 含量折扣)
  B13: 月营销费               (滑条 ¥0~20,000)
  B14: 店铺面积               (滑条 30~300㎡)
  B15: 地段等级               (滑条 1~10, 默认 7)
  B20: 品质分 = B21           (0~1)
  B21: 员工满意度             (0~1, 初始 0.8)
  B24: 员工人数               (滑条 1~20)
  B25: 月培训费/人           (滑条 ¥0~1,000)
  B26: 月工资/人             (滑条 ¥800~10,000, 默认 4,000)
  B22: 库存损耗              (计算量)
  B23: 原料实际进价(旧)

  FAT:    员工疲劳度          (0~100, 初始 40)
  EMP:    员工经验            (0~200, 初始 0)
  BRAND:  品牌值              (0~800)
  TRAFFIC:月客流              (人)
}

=============================================================================
## 二、用户可调参数（滑条）
=============================================================================

  sl[0] → B1   售价          ¥5~40      step 1    默认 16
  sl[1] → B24  员工数         1~20       step 1    默认 8
  sl[2] → B25  培训费/人      ¥0~1,000  step 50   默认 100
  sl[3] → B13  营销费         ¥0~20,000 step 500  默认 2,000
  sl[4] → B10  原料进价       ¥1~6      step 0.1  默认 5
  sl[5] → B14  面积           30~300㎡  step 10   默认 150
  sl[6] → B26  工资/人        ¥800~10K  step 200  默认 4,000
  sl[7] → B15  地段等级       1~10       step 1    默认 7

## 三、单月计算流程 (one_month = rc() + fd() + nx())
=============================================================================

符号说明:
  q.B1  = sl[0].v  (用户的当月决策值)
  s.B1  = c.B1     (状态中的变量, 可跨月累积)

### rc() — 实时计算

#### Step 1: 人工成本 B9
  B9 = round( q.B24 × q.B26 × (1 + q.B15 × 0.15 × max(0, 1 − q.B24 × 0.08)) )

  复杂度系数: 地段越好、人越少 → 人均成本越高(需要更专业)
  例: B15=7, B24=8, B26=4000 → B9 = 8×4000×(1+7×0.15×0.36) = 44,096
      60㎡夫妻店: B15=3, B24=2, B26=3500 → B9 = 2×3500×1.378 = 9,646

#### Step 2: 物理产能 ph 与 实际产量 B3

  面积产能    = q.B14 × 35        (35份/㎡/月)
  人效产能    = q.B24 × 1,500     (1,500份/人/月 ≈ 50份/天)
  产能恢复    = max(0, (2 − s.B4) × 100)  (设备越好恢复越多)

  ph = max(1, min(面积产能, 人效产能) + 产能恢复)

  B3 = round(ph × sk(s.FAT))

  sk(c4) = 1.0         当 c4 < 80
           0.30        当 80 ≤ c4 < 90
           0.10        当 c4 ≥ 90

#### Step 3: 报废成本 B4
  B4 = max(0.1,
           max(0.1, 2 − B3×0.0002) × (1 − EMP×0.002))

  规模效应: 产量越大 → 报废率越低(熟练)
  经验效应: 经验越高 → 报废率越低

#### Step 4: 价格弹性 pb
  pb = 1 + (20 − q.B1) × 0.15      当 q.B1 < 20  (低价引流)
       max(0.6, 1 − (q.B1 − 20) × 0.03)  当 q.B1 ≥ 20  (高价驱客)

  例: ¥15 → pb=1.75  (大幅引流)
      ¥30 → pb=0.70  (客流打7折)

#### Step 5: 客流 TRAFFIC
  base_tr   = round(500 + 500 × q.B15) + round(s.BRAND × 3)
  area_lever = 1 + q.B14 / 100
  mkt_tr    = round( √max(0, q.B13) × 10 )

  tr = round(base_tr × pb) + round(mkt_tr × area_lever)

  地段客流:
    B15=3 → 2,000人/月 (67人/天)
    B15=5 → 3,000人/月 (100人/天)
    B15=7 → 4,000人/月 (133人/天)
    B15=9 → 5,000人/月 (167人/天)

  品牌客流: BRAND×3  (品牌100 → 300人/月)
  营销客流: √营销×10×面积杠杆 (营销¥8,000 → 894人/月)

#### Step 6: 需求 B2 (客单 × 转化率)
  最高可接受价:
    ma = max(1, 15 + q.B15 × 2 + s.BRAND × 0.5 + s.B21 × 3)

    地段B15=3 → ma ≈ 21  (社区顾客消费力弱)
    地段B15=9 → ma ≈ 33  (商场顾客舍得花钱)
    品牌100    → ma +50
    满意度1.0  → ma +3

  转化率:
    当 q.B1 ≤ ma:  conv = 0.5 + (ma − q.B1)/ma × 0.4
    当 q.B1 > ma:  conv = max(0.05, 0.5 × ma / q.B1)

    例: ma=33, 售价¥30 → 转化率 0.5+(33−30)/33×0.4 = 53.6%
        售价¥35 > ma→ 转化率 0.5×33/35 = 47.1%

  B2 = round(tr × conv)

#### Step 7: 房租 B5
  B5 = max(0, round( q.B14 × q.B15 × max(2, 20 − q.B14×0.05) ))

  例: 60㎡, B15=3 → 3,060/月
      100㎡, B15=9 → 13,500/月
      250㎡, B15=7 → 13,125/月

#### Step 8: 员工满意度 B21
  pp = B9 / max(B3, 1)              (人均产值)
  bl = 3 + q.B15 × 0.4              (薪酬基准线)

  薪酬满意度:
    ps = pp/bl × 0.7              当 pp < bl (工资偏低)
        0.7 + min((pp−bl)/(bl×2), 0.3)  当 pp ≥ bl (工资达标)

  B21 = round(min(1,
                  max(0, ps − max(0, B3/ph − (0.8+0.2×ps)) × 1.5)
                 ) × 1000) / 1000

  有效惩罚 = max(0, 产能利用率 − 忍耐阈值) × 1.5
  忍耐阈值 = 0.8 + 0.2×ps  (高薪提高忍耐力)

#### Step 9: 原料成本 B12
  B12 = round( q.B10 × (1 − min(0.5, B3 × 0.00008)) × 100 ) / 100

  量折扣: 产量越大 → 原料进价折扣越大 (上限50%)
  例: B10=5, B3=2,550 → B12 = 5×(1−0.204) = 3.98

#### Step 10: 销量 & 营收
  B6 = min(B2, B3)                    (需求 vs 产能, 取小)

  饮品收入 = round(B6 × q.B1 × 0.20)  (面包营收的20%)
  B7 = B6 × q.B1 + 饮品收入            (总营收, 面包+饮品)

#### Step 11: 成本汇总 & 利润 B8

  COGS/份:
   原料     = B12
   包装     = round(q.B1 × 0.15)       (售价的15%)
   加工损耗 = B4

  总生产成本 = (B12 + 包装 + B4) × B3

  固定月费:
   房租      = B5
   人工      = B9
   营销      = q.B13
   培训      = q.B25 × q.B24
   杂项      = round(0.05 × q.B24 × 1,500)    (社保/福利)
   水电燃气  = round(q.B14 × 25 + q.B24 × 200) (面积费+人头费)
   设备折旧  = 2,000

  B8 = round( B7 − 总生产成本 − 房租 − 人工 − 营销 − 培训 − 杂项 − 水电 − 折旧 )

  TRAFFIC = tr
  B20 = round(B21 × 100) / 100   (品质=满意度)

#### Step 12: 疲劳变化 fd()
  d = 3                        (基线疲劳: 上班就有)
  if 培训=0 且 产能利用率>70%:  d += 5
  if 培训=0:                    d += 2
  if 产能利用率>80%:           d += (利用率−0.8)×40
  d −= 培训费×0.03              (培训降低疲劳)

  if 满意度<0.5:  d −= (满意度−0.5)×15 × (FAT<40 ? FAT/40 : 1)
  if 人均工资>1,500:  d −= (人均工资−1500)×0.005 × (FAT<40 ? FAT/40 : 1)

  FAT = max(0, min(100, round(FAT + d)))

#### Step 13: 跨月更新 nx()
  利润 = B8
  EMP = min(200, round(EMP + max(1, 10 − EMP×0.05)))
       (边际递减: 新员工学得快, 老员工学得慢)

  缺货率 = max(0, (B2−B3)/max(1,B2))
  品牌增 = B21 × 30
  品牌衰 = BRAND × max(0.02, BRAND×0.015)
  增长系数 = max(0.1, 1 − BRAND/800)  (品牌天花板800)

  BRAND = max(0, round(BRAND + 品牌增×增长系数 − 品牌衰 − 缺货率×10))
  FAT = nf
